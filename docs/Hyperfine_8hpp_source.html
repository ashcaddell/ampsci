29 inline auto volotkaBW_F(double mu, double I_nuc, double l_pn, int gl)
30  -> Func_R2_R
31 // Function that returns generates + returns F_BW Bohr-Weiskopf
32 // gl = 1 for proton, =0 for neutron. Double allowed for testing..
33 // mu is in units of Nuclear Magneton!
34 {
35  const auto two_I = int(2 * I_nuc + 0.0001);
36  const auto two_l = int(2 * l_pn + 0.0001);
37  const auto g_l = double(gl); // just safety
38  const auto gI = mu / I_nuc;
39 
40  const auto K = (l_pn * (l_pn + 1.0) - (3. / 4.)) / (I_nuc * (I_nuc + 1.0));
41  const double g_s = (2.0 * gI - g_l * (K + 1.0)) / (1.0 - K);
42  std::cout << "BW using: gl=" << g_l << ", gs=" << g_s << ", l=" << l_pn
43  << ", gI=" << gI << " (I=" << I_nuc << ")\n";
44  const double factor =
45  (two_I == two_l + 1) ?
46  g_s * (1 - two_I) / (4.0 * (two_I + 2)) + g_l * 0.5 * (two_I - 1) :
47  g_s * (3 + two_I) / (4.0 * (two_I + 2)) +
48  g_l * 0.5 * two_I * (two_I + 3) / (two_I + 2);
49  if (two_I != two_l + 1 && two_I != two_l - 1) {
50  std::cerr << "\nFAIL:59 in Hyperfine (volotkaBW_F):\n "
51  "we must have I = l +/- 1/2, but we have: I,l = "
52  << I_nuc << "," << l_pn << "\n";
53  return [](double, double) { return 0.0; };
54  }
55  return [=](double r, double rN) {
56  return (r > rN) ? 1.0 :
57  ((r * r * r) / (rN * rN * rN)) *
58  (1.0 - (3.0 / mu) * std::log(r / rN) * factor);
59  };
60 }

63 inline auto doublyOddBW_F(double mut, double It, double mu1, double I1,
64  double l1, int gl1, double I2, double l2) -> Func_R2_R
65 // F(r) * g = 0.5 [ g1F1 + g2F2 + (g1F1 - g2F2) * K]
66 // K = [I1(I1+1) - I2(I2+1)] / [I(I+1)]
67 // return F(r) [divide by g]
68 // volotkaBW_F(mu, I, l, g_l); //gl is 1 or 0
69 // g2 : from: g = 0.5 [ g1 + g2 + (g1 - g2) * K]
70 {
71  const auto K = (I1 * (I1 + 1.0) - I2 * (I2 + 1.0)) / (It * (It + 1.0));
72  const auto gt = mut / It;
73  const auto g1 = mu1 / I1;
74  const auto g2 = (g1 * (K + 1.0) - 2.0 * gt) / (K - 1.0);
75  const auto mu2 = g2 * I2;
76  const auto gl2 = (gl1 == 0) ? 1 : 0;
77  const auto F1 = volotkaBW_F(mu1, I1, l1, gl1);
78  const auto F2 = volotkaBW_F(mu2, I2, l2, gl2);
79  return [=](double r, double rN) {
80  return (0.5 / gt) * (g1 * F1(r, rN) + g2 * F2(r, rN) +
81  K * (g1 * F1(r, rN) - g2 * F2(r, rN)));
82  };
83 }

86  const Func_R2_R &hfs_F) {

88  rfunc.reserve(rgrid.num_points());

93 inline std::vector<double> RadialFunc(int k, double, const Grid &rgrid,
94  const std::vector<double> &hfs_F) {
95  std::vector<double> rfunc = hfs_F;
96  for (auto ir = 0ul; ir < hfs_F.size(); ir++) {
97  if (rfunc[ir] != 0.0) {
98  rfunc[ir] /= pow(rgrid.r(ir), k + 1);
99  } else {
100  rfunc[ir] = 1.0 / pow(rgrid.r(ir), k + 1);
101  }
102  }
103  return rfunc;
